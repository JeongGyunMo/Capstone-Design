package com.example.clubmanagement.Restaurant;

import android.content.Intent;
import android.os.Bundle;
import android.support.v7.app.AppCompatActivity;
import android.view.View;
import android.widget.Button;
import android.widget.ListView;

import com.example.clubmanagement.Adapter.ListViewAdapter;
import com.example.clubmanagement.Database.RestaurantData;
import com.example.clubmanagement.R;
import com.example.clubmanagement.ClubPage.Club_page;

import java.util.ArrayList;
import java.util.HashMap;

public class Restaurant_Activity extends AppCompatActivity {
    Button back;
    private ListView listview;
    private ListViewAdapter adapter;
    HashMap<String, String> Restaurant = new HashMap<String, String>();
    ArrayList<HashMap<String, String>> Restaurant_list;
    RestaurantData RD = new RestaurantData();

    public void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.restaurant_page);

        DataInput();

        back = (Button) findViewById(R.id.back);
        back.setOnClickListener(new Button.OnClickListener() {
            @Override
            public void onClick(View v) {
                startActivity(new Intent(getApplicationContext(), Club_page.class));
            }
        });
    }

    private void DataInput(){
        listview.setAdapter(adapter);
        RD.ClearListData();
        RD.GetListData(RD.Temp);
        Restaurant_list = RD.Restaurant_list;

        for (int i = 0; i < Restaurant_list.size(); i++) {
            Restaurant = Restaurant_list.get(i);
        }
    }
}
